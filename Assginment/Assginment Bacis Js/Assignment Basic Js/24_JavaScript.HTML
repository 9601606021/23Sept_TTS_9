24. Write to check whether a number is negative, positive or zero?

Program :


            <h3 id="ans1"></h3>
            <h3 id="ans2"></h3>
            <h3 id="ans3"></h3>

            <script>
                const number = prompt("Enter a Number = ");
                if (number > 0) {
                    document.getElementById("ans1").innerHTML =(number +  " is POSITIVE NUMBER");
                }
                else if (number < 0) {
                    document.getElementById("ans2").innerHTML =( number +  " is NEGATIVE NUMBER ");
                }
                else {
                    document.getElementById("ans3").innerHTML =(number +  " is ZERO NUMBER");
                };
            </script>
